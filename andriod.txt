    <script>
        const stations = [
            { id: 1, x: 25, y: 35, title: "台東市中央市場", history: "台東市中央市場的根源可追溯至日治時期，現今規模成形於1963年。它是東台灣最大的零售物資集散中心，更是台東人的靈魂胃袋。" },
            { id: 2, x: 75, y: 25, title: "寶町藝文中心", history: "1936年落成，原為台東街役場高級官員宿舍，其中規模最大者為街長官舍。2004年整建為藝文中心。" },
            { id: 3, x: 45, y: 55, title: "台東中華會館", history: "1927年落成，是當時台東華僑團結力量與抗日聯絡的核心，見證了著名的中華會館事件。" },
            { id: 4, x: 15, y: 75, title: "大同市場攤販部 (戲院遺址)", history: "原為1958年開幕之「大同戲院」附屬建築。戲院帶動了周邊商圈，形成繁華的生活核心。2009年大火後戲院歇業，此處成為時光遺落的角落。" },
            { id: 5, x: 65, y: 80, title: "台東觀光夜市", history: "正氣路最早是水果街。2010年轉型為觀光夜市，實現「晝則水果街，夜則觀光市」的時空切換。" }
        ];

        let audioUnlocked = false;
        let selectedVoice = null;

        function loadVoices() {
            const voices = window.speechSynthesis.getVoices();
            if (voices.length > 0) {
                selectedVoice = voices.find(v => v.lang === 'zh-TW') || 
                                voices.find(v => v.lang.includes('zh-HK')) || 
                                voices.find(v => v.lang.includes('zh'));
            }
        }

        window.speechSynthesis.onvoiceschanged = loadVoices;

        function unlockAudio(showGreeting = false) {
            window.speechSynthesis.cancel();
            
            // 立即發送一個靜音字串來喚醒音訊上下文
            const silent = new SpeechSynthesisUtterance('啟動中');
            silent.volume = 0;
            window.speechSynthesis.speak(silent);
            
            audioUnlocked = true;
            const btn = document.getElementById('audio-unlock-btn');
            btn.className = "bg-green-600 text-white px-6 py-3 rounded-full font-bold shadow-lg transition-all active:scale-95 flex items-center";
            btn.innerHTML = '<i class="fa-solid fa-check-circle mr-2"></i> 語音就緒';
            
            document.getElementById('android-tip').style.display = 'none';

            if (showGreeting) {
                setTimeout(() => {
                    playTour('導覽系統已成功連線。請點擊地圖標記，開始您的老台東旅程。');
                }, 300);
            }
        }

        function initMarkers() {
            const container = document.getElementById('markers-container');
            container.innerHTML = "";
            stations.forEach(s => {
                const el = document.createElement('div');
                el.className = 'station-marker';
                el.style.left = s.x + '%';
                el.style.top = s.y + '%';
                el.innerHTML = `
                    <div class="relative group">
                        <div class="ripple"></div>
                        <div class="w-12 h-12 bg-stone-800 rounded-full border-4 border-white shadow-lg flex items-center justify-center text-white font-black text-lg relative z-20 group-hover:bg-orange-600 transition-colors">
                            ${s.id}
                        </div>
                    </div>
                `;
                el.onclick = () => selectStation(s);
                container.appendChild(el);
            });
        }

        function selectStation(s) {
            if (!audioUnlocked) {
                document.getElementById('android-tip').style.display = 'flex';
                // 跳動效果提醒啟動按鈕
                document.getElementById('audio-unlock-btn').classList.add('animate-bounce');
                setTimeout(() => document.getElementById('audio-unlock-btn').classList.remove('animate-bounce'), 2000);
                return;
            }

            const content = document.getElementById('detail-content');
            content.innerHTML = `
                <div class="animate-fadeIn">
                    <span class="text-xs font-black text-orange-600 tracking-[0.3em] uppercase mb-2 block">據點檔案 ${s.id}</span>
                    <h2 class="text-3xl font-black text-stone-800 mb-6 border-l-8 border-orange-600 pl-4">${s.title}</h2>
                    <p class="text-lg text-stone-600 leading-relaxed mb-10">${s.history}</p>
                    <div class="sticky bottom-0 bg-white/95 backdrop-blur-sm py-4">
                        <button onclick="playTour('${s.title}。${s.history}')" class="w-full bg-stone-800 text-white py-5 rounded-2xl font-bold shadow-lg flex items-center justify-center gap-3 active:scale-95 transition-all">
                            <i class="fa-solid fa-volume-high"></i> 再次收聽
                        </button>
                    </div>
                </div>
            `;
            playTour(`${s.title}。${s.history}`);
        }

        function playTour(text) {
            if(!audioUnlocked) return;
            window.speechSynthesis.cancel();
            
            setTimeout(() => {
                const utterance = new SpeechSynthesisUtterance(text);
                utterance.lang = 'zh-TW';
                utterance.rate = 0.9; // 稍微放慢語速更有導覽感
                
                if(!selectedVoice) loadVoices();
                if(selectedVoice) utterance.voice = selectedVoice;

                utterance.onerror = (e) => {
                    console.error("TTS Error:", e);
                    // 如果在 GitHub/HTTPS 環境下出錯，嘗試重置
                    if (e.error === 'not-allowed') audioUnlocked = false;
                };

                window.speechSynthesis.speak(utterance);
            }, 100);
        }

        function toggleAIChat() { document.getElementById('ai-chat').classList.toggle('hidden'); }

        function askAI() {
            const input = document.getElementById('chat-input');
            const chatBox = document.getElementById('chat-box');
            if(!input.value.trim()) return;
            chatBox.innerHTML += `<div class="flex justify-end"><div class="bg-stone-800 text-white p-3 rounded-2xl max-w-[85%]">${input.value}</div></div>`;
            input.value = "";
            chatBox.scrollTop = chatBox.scrollHeight;
        }

        window.onload = () => {
            initMarkers();
            loadVoices();
            if (/Android/i.test(navigator.userAgent)) {
                document.getElementById('android-tip').style.display = 'flex';
            }
        };
    </script>
